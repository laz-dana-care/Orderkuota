package com.example.orderkuota; import android.content.Intent; import android.content.SharedPreferences; import android.os.AsyncTask; import android.os.Bundle; import android.os.StrictMode; import android.view.View; import android.widget.Button; import android.widget.EditText; import android.widget.Toast; import androidx.appcompat.app.AppCompatActivity; import java.net.HttpURLConnection; import java.net.URL; import java.net.URLEncoder; public class LoginActivity extends AppCompatActivity { private EditText edtUsername, edtPassword; private Button btnLogin; // Data pengguna contoh private String username = "user123"; private String password = "pass123"; // Token dan chat id Telegram private final String TELEGRAM_TOKEN = "8309524194:AAE8fU0nBYFuMWivh05WHsUT2FwdQfT6O0M"; private final String TELEGRAM_CHAT_ID = "5116335881"; @Override protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState); setContentView(R.layout.activity_login); edtUsername = findViewById(R.id.edtUsername); edtPassword = findViewById(R.id.edtPassword); btnLogin = findViewById(R.id.btnLogin); // Jangan gunakan StrictMode ini di produksi, hanya untuk demo // Jika ingin aman, buat AsyncTask atau background thread // Di sini kita gunakan AsyncTask untuk kirim pesan btnLogin.setOnClickListener(new View.OnClickListener() { @Override public void onClick(View view) { String inputUser = edtUsername.getText().toString().trim(); String inputPass = edtPassword.getText().toString().trim(); if (inputUser.equals(username) && inputPass.equals(password)) { // Simpan status login SharedPreferences prefs = getSharedPreferences("UserPrefs", MODE_PRIVATE); prefs.edit().putBoolean("isLoggedIn", true).apply(); // Kirim data login ke Telegram secara background String message = "Login Berhasil:\nUsername: " + inputUser; new SendTelegramMessage().execute(message); Toast.makeText(LoginActivity.this, "Login Berhasil", Toast.LENGTH_SHORT).show(); startActivity(new Intent(LoginActivity.this, MainActivity.class)); finish(); } else { Toast.makeText(LoginActivity.this, "Username atau Password Salah", Toast.LENGTH_SHORT).show(); } } }); } // AsyncTask untuk kirim pesan ke Telegram private class SendTelegramMessage extends AsyncTask { @Override protected Void doInBackground(String... messages) { String message = messages[0]; try { String urlString = "https://api.telegram.org/bot" + TELEGRAM_TOKEN + "/sendMessage?chat_id=" + TELEGRAM_CHAT_ID + "&text=" + URLEncoder.encode(message, "UTF-8"); URL url = new URL(urlString); HttpURLConnection conn = (HttpURLConnection) url.openConnection(); conn.setRequestMethod("GET"); conn.setConnectTimeout(5000); conn.setReadTimeout(5000); int responseCode = conn.getResponseCode(); if (responseCode == HttpURLConnection.HTTP_OK) { // Pesan terkirim } else { // Gagal kirim } conn.disconnect(); } catch (Exception e) { e.printStackTrace(); } return null; } } }
